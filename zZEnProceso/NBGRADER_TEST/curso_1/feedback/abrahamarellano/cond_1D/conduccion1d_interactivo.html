def conduccion_1d(k, S, L, TA, TB, N):
    """
    Parameters
    ---------
    L: Float
    es la longitud del dominio
    
    k: float
    es la conductividad del material
    
    S: float 
    es la fuente 
     
    TA: float
    Es la condicin de frontera a la izquierda.

    TB:
    Es la condicin de frontera a la dereccha.

    N = int
    Es el nmero de nodos internos (grados de libertad).
    
    Return
    ------
    Al final esta funcin dibuja la solucin.

    """
    h = L/(N+1)

    T = np.zeros(N+2)
    b = np.zeros(N)
    A = np.zeros((N,N))

    r = k/h**2
    b[:] = S/r
    T[0] = TA
    T[-1] = TB
    b[0] += TA
    b[-1] += TB

    A[0,0] = 2
    A[0,1] = -1
    A[-1,-2] = -1
    A[-1,-1] = 2

    for i in range(1,N-1):
        A[i,i] = 2
        A[i,i+1] = -1
        A[i,i-1] = -1
    
    T[1:N+1] = np.linalg.solve(A,b)
    plt.plot(T)
